# --- MOVIES ---
# List (Success)
curl -X GET "http://localhost:5000/api/v1/movies?page=1&size=10"
# Read (Success)
curl -X GET "http://localhost:5000/api/v1/movies/1"
# Error 404 (Not Found)
curl -X GET "http://localhost:5000/api/v1/movies/999"

# --- USERS ---
# Create (Success)
curl -X POST "http://localhost:5000/api/v1/users" -H "Content-Type: application/json" -d "{\"name\": \"Jeremie\", \"email\": \"jeremie@example.com\"}"
# Error 400 (Bad Request - Missing Email)
curl -X POST "http://localhost:5000/api/v1/users" -H "Content-Type: application/json" -d "{\"name\": \"Jeremie\"}"

# --- ACTORS ---
# List (Success)
curl -X GET "http://localhost:5000/api/v1/actors"
# Delete (Success)
curl -X DELETE "http://localhost:5000/api/v1/actors/5"
